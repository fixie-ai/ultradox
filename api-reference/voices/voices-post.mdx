---
openapi: post /api/voices
title: "Create (Clone) Voice"
description: "Create a new cloned voice"
---

Any created voices are private to your account.

Uses multipart/form-data encoding to provide the name of the voice along with an audio file containing the voice to be used for cloning.


<RequestExample>

```bash cURL
curl --request GET \
  --url https://api.ultravox.ai/api/voices \
  --header 'Content-Type: multipart/form-data' \
  --header "X-API-Key: <your_api_key> \
  --form 'file=@/path/to/file.mp3' \
  --form 'name=<voice_name>'
```

```go Go
package main

import (
	"fmt"
	"strings"
	"net/http"
	"io/ioutil"
)

func main() {

	url := "https://api.ultravox.ai/api/voices"

	payload := strings.NewReader("-----011000010111000001101001--\r\n\r\n")

	req, _ := http.NewRequest("POST", url, payload)

	req.Header.Add("X-API-Key", "<api-key>")
	req.Header.Add("Content-Type", "multipart/form-data")

	res, _ := http.DefaultClient.Do(req)

	defer res.Body.Close()
	body, _ := ioutil.ReadAll(res.Body)

	fmt.Println(res)
	fmt.Println(string(body))

}
```
